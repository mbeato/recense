 * before insightReflector.reflect()). Derives:

async deriveDocGraph(): Promise<void> {
  // ── Phase A: async-free read-only ────────────────────────────────────
  // 1. Load all doc nodes + their subject-schema-ids meta
  // 2. For each subject doc: expand its schema member set (D-37 gated)
  // 3. Compute IDF per member across corpus (D-03)
  // 4. For each doc pair: compute weighted shared-member score (D-01/D-03)
  // 5. Add schema_rel adjacency weight (D-02)
  // 6. top-K cutoff per doc (D-03), union symmetry (D-05)
  // 7. Containment: resolve abstracts-ladder ancestry per schema set (D-06/D-09)
  //    Hub→subject edges: every subject's hub gets a containment edge (D-08)

  // 1. stmtWipeDocContainment.run()
  // 2. stmtWipeDocReference.run()
  // 3. Write new doc_containment edges via store.upsertEdge (CONSOL-03)
  // 4. Write new doc_reference edges via store.upsertEdge (CONSOL-03)

**Atomic transaction pattern** (schema-relations.ts L355–375):
```typescript
// T-02-ASYNC: NO await inside the transaction (all async / reads done in Phase A above)
this.db.transaction(() => {
  // D-04 wipe-from-scratch: delete all prior derived artifacts (idempotent rebuild)
  this.stmtWipeDocContainment.run();
  this.stmtWipeDocReference.run();

  // Write new edges via owned primitives (CONSOL-03 — no raw INSERT on edge table)
  for (const { srcId, dstId, w } of containmentEdges) {
    this.store.upsertEdge({
      src: srcId,
      dst: dstId,
      rel: 'doc_containment',
      kind: 'doc_containment',
      w,
      last_access: nowMs,
    });
  }

  for (const { srcId, dstId, w } of referenceEdges) {
    // Stable ordering (lexicographic by doc id) for deterministic idempotency
    const [src, dst] = srcId < dstId ? [srcId, dstId] : [dstId, srcId];
    this.store.upsertEdge({
      src: src!,
      dst: dst!,
      rel: 'doc_reference',
      kind: 'doc_reference',
      w,
      last_access: nowMs,
    });
  }
}).immediate(); // M-5: write-lock discipline — avoid SQLITE_BUSY_SNAPSHOT in WAL mode
```

async function main(): Promise<void> {
  const argv = process.argv;

  // No positional arg — operates on all docs in DB.

  const dbPath = resolveSharedDbPath(argv, { fallbackToDefault: false });

  // ── 3. Acquire the shared write lock (T-25-08) ─────────────────────────────
  if (!acquireLock()) {
    process.stderr.write('recense derive-doc-graph: Lock held by another process — exiting\n');
    process.exit(0);

  let db: Database.Database | undefined;
  try {
    fileLog(`starting: dbPath=${dbPath} dryRun=${isDryRun}`);

    db = new Database(dbPath);
    db.pragma('foreign_keys = ON');
    initSchema(db);

    const config = { ...DEFAULT_CONFIG, dbPath };
    const store = new SemanticStore(db, realClock, config);

    // ── 4. Construct DocGraphDeriver ────────────────────────────────────────
    const deriver = new DocGraphDeriver(db, store, config, realClock);

    fileLog(`running deriveDocGraph (dryRun=${isDryRun})`);

    // ── 5. Run the derivation pass ──────────────────────────────────────────
    const result = await deriver.deriveDocGraph({ dryRun: isDryRun });

    fileLog(
      `done: containment=${result.containment} reference=${result.reference}`,
    );

    // ── 6. Emit result JSON ─────────────────────────────────────────────────
    process.stdout.write(
      JSON.stringify({
        containment: result.containment,
        reference: result.reference,
        dryRun: isDryRun,
      }) + '\n',
    );

    releaseLock();
    process.exit(1);
  });
}
